def first_name_middle_initial
  pieces = []
  (pieces << first_name)           if first_name?
  (pieces << middle_name[0...1])   if middle_name?
  return pieces.join(' ')

def list_name
  pieces = []
  (pieces << first_name)  if first_name?
  (pieces << middle_name) if middle_name?
  if last_name?
    comma = pieces.size>0 ? ',' : ''
    pieces.unshift(last_name+comma)
  end
  return pieces.join(' ')
end
