def full_name
  pieces = []
  if given_name?
    pieces << given_name
  elsif first_name?
    pieces << first_name
  end
  if middle_name?
    pieces << middle_name
  end
  if family_name?
    pieces << family_name
  elsif last_name?
    pieces << last_name
  end
  return pieces.join(' ')
end

def list_name
  left_pieces = []
  right_pieces = []
  if family_name?
    left_pieces << family_name
  elsif last_name?
    left_pieces << last_name
  end
  if given_name?
    right_pieces << given_name
  elsif first_name?
    right_pieces << first_name
  end
  if middle_name?
    right_pieces << middle_name
  end
  return left_pieces.join(' ') + (!left_pieces.empty? && !right_pieces.empty? ? ", " : '') + right_pieces.join(' ')
end
